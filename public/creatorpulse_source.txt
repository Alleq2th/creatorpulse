    .replace(/&/g, '&amp;')

async function apiFetch(endpoint, options) {
  options = options || {};
  try {
    const url = API + endpoint + (endpoint.includes('?') ? '&' : '?') + '_=' + Date.now();
    const r = await fetch(url, Object.assign({ cache: 'no-store' }, options));
    if (!r.ok) {
      const err = await r.text().catch(function() { return 'Unknown error'; });
      throw new Error('HTTP ' + r.status + ': ' + err);
    }
    return await r.json();
  } catch (e) {
    console.error('API Error [' + endpoint + ']:', e.message);
    throw e;
  }
}

function restoreSession() {
  try {
    const saved = localStorage.getItem('cp_session');
    if (!saved) return false;
    const parsed = JSON.parse(saved);
    if (parsed.token && parsed.user && parsed.user.email) {
      AUTH_TOKEN = parsed.token;
      USER = parsed.user;
      AUTH_MODE = 'app';
      return true;
    }
  } catch (e) {
    console.warn('Session restore failed:', e);
    clearSession();
  }
  return false;
}

function saveSession(token, user) {
  try {
    localStorage.setItem('cp_session', JSON.stringify({ token: token, user: user }));
  } catch (e) {
    console.warn('Failed to save session:', e);
  }
}

function clearSession() {
  try { localStorage.removeItem('cp_session'); } catch (e) {}
  AUTH_TOKEN = null;
  USER = null;
  AUTH_MODE = 'landing';
}

function rr(ctx, x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x + r, y);
  ctx.lineTo(x + w - r, y);
  ctx.quadraticCurveTo(x + w, y, x + w, y + r);
  ctx.lineTo(x + w, y + h - r);
  ctx.quadraticCurveTo(x + w, y + h, x + w - r, y + h);
  ctx.lineTo(x + r, y + h);
  ctx.quadraticCurveTo(x, y + h, x, y + h - r);
  ctx.lineTo(x, y + r);
  ctx.quadraticCurveTo(x, y, x + r, y);
  ctx.closePath();
}

function wrapText(ctx, text, maxChars) {
  if (!text) return [];
  const words = text.split(' ');
  const lines = [];
  var cur = '';
  for (var i = 0; i < words.length; i++) {
    var w = words[i];
    if (w.length > maxChars) {
      if (cur) { lines.push(cur); cur = ''; }
      for (var j = 0; j < w.length; j += maxChars) {
        lines.push(w.slice(j, j + maxChars));
      }
      continue;
    }
    var t = cur ? cur + ' ' + w : w;
    if (t.length > maxChars) {
      lines.push(cur);
      cur = w;
    } else {
      cur = t;
    }
  }
  if (cur) lines.push(cur);
  return lines;
}

function makeThumbnailCanvas(headline, subjectImg, channel, accent) {
  var W = 1280, H = 720;

    try {
      ctx.save();
      var iw = subjectImg.naturalWidth || subjectImg.width || 600;

async function loadSubjectImage(headline, niche, articleImageUrl) {
  if (articleImageUrl) {
    var url = API + '/api/image-proxy?url=' + encodeURIComponent(articleImageUrl);
    var img = await loadImageElement(url);
    if (img) return img;
  }
  var sportKeywords = ['football', 'soccer', 'basketball', 'tennis', 'cricket', 'boxing', 'mma', 'nba', 'nfl', 'premier league', 'champions league', 'f1'];
  var isSport = sportKeywords.some(function(k) { return (niche + headline).toLowerCase().includes(k); });
  var words = headline.split(' ');
  var properNouns = words.filter(function(w) { return w.length > 2 && w[0] === w[0].toUpperCase() && w[0] !== w[0].toLowerCase(); });
  var subject = properNouns.slice(0, 2).join(' ');
  if (!subject) return null;
  try {
    var imgUrl = null;
    if (isSport) {
      imgUrl = await fetchSportsImage(subject, 'player');
      if (!imgUrl) imgUrl = await fetchSportsImage(subject, 'team');
    }
    if (!imgUrl) imgUrl = await fetchWikiImage(subject);
    if (!imgUrl) return null;
    return await loadImageElement(imgUrl);
  } catch (e) { return null; }
}

async function fetchSportsImage(query, type) {
  try {
    var d = await apiFetch('/api/sports-image?query=' + encodeURIComponent(query) + '&type=' + type);
    return d.image || null;
  } catch (e) { return null; }
}

async function fetchWikiImage(query) {
  try {
    var d = await apiFetch('/api/wiki-image?query=' + encodeURIComponent(query));
    return d.image || null;
  } catch (e) { return null; }
}

function loadImageElement(url) {
  return new Promise(function(resolve) {
    var img = new Image();
    img.crossOrigin = 'anonymous';
    img.onload = function() { resolve(img); };
    img.onerror = function() { resolve(null); };
    img.src = url;
    setTimeout(function() { resolve(null); }, 6000);
  });
}

    var niche = USER.niches[i];
    try { var articles = await fetchNews(niche); ALL_TRENDS.push.apply(ALL_TRENDS, articles); } catch (e) { console.warn('news fetch failed for', niche); }
    try { var blogArticles = await fetchBlogFeed(niche); ALL_TRENDS.push.apply(ALL_TRENDS, blogArticles); } catch (e) {}
    var handles = nicheHandles[niche] || nicheHandles['default'];
    for (var j = 0; j < handles.length; j++) {
      try {
        var articles = await fetchTwitterFeed(handles[j]);
        articles.forEach(function(a) {
          a.niche = niche;
          if (a.image && !a.image.startsWith('http')) a.image = 'https://nitter.net' + a.image;
        });
        ALL_TRENDS.push.apply(ALL_TRENDS, articles);
      } catch (e) {}

  if (USER && USER.niches) {
    var eventPromises = USER.niches.map(function(n) { return fetchEvents(n); });
    var results = await Promise.all(eventPromises);
    results.forEach(function(events) { allEvents = allEvents.concat(events || []); });
    var seen = new Set();
    allEvents = allEvents.filter(function(e) {
      var key = (e.title || '') + (e.date || '');
      if (seen.has(key)) return false;
      seen.add(key);
      return true;
    });
  }
  var html = '<div class="card"><div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:8px"><span style="font-weight:700;font-size:15px">' + MONTHS[CAL_M] + ' ' + CAL_Y + '</span><div><button class="btn bs bsm" id="cal-prev">‹</button><button class="btn bs bsm" id="cal-next">›</button></div></div><div class="cal-grid">';
  DAYS.forEach(function(d) { html += '<div class="cal-day">' + d + '</div>'; });
  for (var i = 0; i < first; i++) html += '<div></div>';
  for (var d = 1; d <= days; d++) {
    var isTod = isCur && d === today;
    var hasPost = postsByDate[d] && postsByDate[d].length > 0;
    var hasEvent = allEvents.some(function(e) {
      try { return new Date(e.date).getDate() === d && new Date(e.date).getMonth() === CAL_M && new Date(e.date).getFullYear() === CAL_Y; } catch (err) { return false; }
    });
    var isSel = SELECTED_DAY === d;
    var cls = 'cal-cell ' + (isTod ? 'today ' : '') + ((hasPost || hasEvent) ? 'has-event ' : '') + (isSel ? 'selected' : '');
    html += '<div class="' + cls.trim() + '" data-day="' + d + '"><div class="day-num">' + d + '</div>' + (hasEvent ? '<div class="cal-event-dot"></div>' : '') + (hasPost ? '<div class="cal-post-dot"></div>' : '') + '</div>';
  }
  html += '</div></div>';
  html += '<div class="card"><div class="card-title">📌 Upcoming Events</div>';
  var combined = [
    ...allEvents.map(function(e) { return { date: e.date, title: e.title, type: '📅 ' + (e.title || 'Event'), desc: e.description || '' }; }),
    ...SCHEDULE_DATA.map(function(p) { return { date: p.scheduled_date, title: p.headline, type: '📌 ' + (p.headline || 'Post'), desc: p.niche || '' }; })
  ];
  combined.sort(function(a, b) {
    try { return new Date(a.date || 0) - new Date(b.date || 0); } catch (e) { return 0; }
  });
  var future = combined.filter(function(item) {
    try { return new Date(item.date || 0) >= now; } catch (e) { return true; }
  });
  if (future.length === 0) {
    html += '<div style="font-size:12px;color:var(--mu)">No upcoming events or scheduled posts.</div>';
  } else {
    future.slice(0, 6).forEach(function(item) {
      html += '<div class="event-item"><div class="event-date">' + escapeHTML(item.date || 'Draft') + '</div><div class="event-title">' + escapeHTML(item.type) + '</div>';
      if (item.desc) html += '<div class="event-desc">' + escapeHTML(item.desc) + '</div>';
      html += '</div>';
    });
  }
  html += '</div>';
  container.innerHTML = html;
}

// EVENT HANDLERS (Event Delegation)
// ═══════════════════════════════════════════════════════════════════
function initEventDelegation() {
  var homeContent = document.getElementById('home-content');
  if (homeContent) {
    homeContent.addEventListener('click', async function(e) {
      var target = e.target;
      var th = target.closest('.th');
      if (th) {
        var tid = th.dataset.tid;
        if (tid) { await toggleCard(tid); return; }
      }
      var ptab = target.closest('.ptab');
      if (ptab) {
        var tid = ptab.dataset.tid;
        var pid = ptab.dataset.pid;
        if (tid && pid) {
          AP[tid] = pid;
          AT[tid] = (PTYPES[pid] && PTYPES[pid][0]) || '';
          renderHome();
          return;
        }
      }
      var ctp = target.closest('.ctp');
      if (ctp) {
        var tid = ctp.dataset.tid;
        var ct = ctp.dataset.ct;
        if (tid && ct) { AT[tid] = ct; renderHome(); return; }
      }
      var genBtn = target.closest('.gen-btn');
      if (genBtn) {
        var tid = genBtn.dataset.tid;
        if (tid && !genBtn.disabled) { await gen(tid, false); return; }
      }
      var genAllBtn = target.closest('.gen-all-btn');
      if (genAllBtn) {
        var tid = genAllBtn.dataset.tid;
        if (tid && !genAllBtn.disabled) { await gen(tid, true); return; }
      }
      var copyBtn = target.closest('.cb[data-copy]');
      if (copyBtn) {
        e.stopPropagation();
        var text = copyBtn.dataset.copy;
        if (text) {
          navigator.clipboard.writeText(text).then(function() {
            copyBtn.textContent = '✓';
            copyBtn.classList.add('copied');
            setTimeout(function() { copyBtn.textContent = 'Copy'; copyBtn.classList.remove('copied'); }, 1500);
          }).catch(function() { showToast('❌ Copy failed', 'error'); });
        }
        return;
      }
      var dlBtn = target.closest('.dl-btn');
      if (dlBtn) {
        e.stopPropagation();
        var url = dlBtn.dataset.url;
        var name = dlBtn.dataset.name;
        if (url) {
          var a = document.createElement('a');
          a.href = url;
          a.download = name || 'download.png';
          document.body.appendChild(a);
          a.click();
          document.body.removeChild(a);
          showToast('⬇ Downloaded!', 'success');
        }
        return;
      }
      var saveBtn = target.closest('.save-output-btn');
      if (saveBtn) {
        e.stopPropagation();
        var tid = saveBtn.dataset.tid;
        var idx = parseInt(saveBtn.dataset.index);
        if (tid && !isNaN(idx)) { await saveOutputToCalendar(tid, idx); return; }
      }
      var previewImg = target.closest('.preview-img');
      if (previewImg) {
        e.stopPropagation();
        openImageModal(previewImg.src);
        return;
      }
    });
  }

  var savedContent = document.getElementById('saved-content');
  if (savedContent) {
    savedContent.addEventListener('click', function(e) {
      var target = e.target;
      var calBtn = target.closest('.cal-export-btn');
      if (calBtn) {
        e.stopPropagation();
        try {
          var post = JSON.parse(calBtn.dataset.post);
          openGoogleCalendar(post);
        } catch (err) { showToast('❌ Error opening calendar', 'error'); }
        return;
      }
      var delBtn = target.closest('.delete-post-btn');
      if (delBtn) {
        e.stopPropagation();
        if (!confirm('Delete this saved post?')) return;
        var postId = delBtn.dataset.postid;
        deletePost(postId);
        return;
      }
      var viewBtn = target.closest('.view-post-btn');
      if (viewBtn) {
        e.stopPropagation();
        var postId = viewBtn.dataset.postid;
        var post = SAVED_POSTS.find(function(p) { return p.id === postId; });
        if (post) viewSavedPost(post);
        return;
      }
      var copyBtn = target.closest('.copy-post-btn');
      if (copyBtn) {
        e.stopPropagation();
        var postId = copyBtn.dataset.postid;
        var post = SAVED_POSTS.find(function(p) { return p.id === postId; });
        if (post && post.content) {
          var text = post.content.content || post.content.hook || JSON.stringify(post.content);
          navigator.clipboard.writeText(text).then(function() { showToast('📋 Copied!', 'success'); }).catch(function() { showToast('❌ Copy failed', 'error'); });
        }
        return;
      }
      var shareBtn = target.closest('.share-post-btn');
      if (shareBtn) {
        e.stopPropagation();
        var postId = shareBtn.dataset.postid;
        var post = SAVED_POSTS.find(function(p) { return p.id === postId; });
        if (post && navigator.share) {
          navigator.share({ title: post.headline, text: post.content && post.content.content ? post.content.content.slice(0, 200) : post.headline }).catch(function() {});
        } else {
          showToast('🔗 Web Share API not supported', 'warning');
        }
        return;
      }
    });
  }

  var calendarContent = document.getElementById('calendar-content');
  if (calendarContent) {
    calendarContent.addEventListener('click', function(e) {
      var target = e.target;
      var prevBtn = target.closest('#cal-prev');
      if (prevBtn) {
        if (CAL_M === 0) { CAL_M = 11; CAL_Y--; } else { CAL_M--; }
        loadSchedule().then(function() { renderCalendar(); });
        return;
      }
      var nextBtn = target.closest('#cal-next');
      if (nextBtn) {
        if (CAL_M === 11) { CAL_M = 0; CAL_Y++; } else { CAL_M++; }
        loadSchedule().then(function() { renderCalendar(); });
        return;
      }
      var dayCell = target.closest('.cal-cell[data-day]');
      if (dayCell) {
        var day = parseInt(dayCell.dataset.day);
        SELECTED_DAY = SELECTED_DAY === day ? null : day;
        renderCalendar();
        if (SELECTED_DAY !== null) showDayDetail(day);
        return;
      }
    });
  }
}

async function saveOutput(t, output, ap) {
  if (!output || !USER || !AUTH_TOKEN) return;
  var contentType = 'Script';
  if (output.type === 'carousel') contentType = 'Carousel';
  else if (output.type === 'thumbnail') contentType = 'Thumbnail';
  else if (output.type === 'noface') contentType = 'Graphic';
  var today = new Date();
  var scheduledDate = today.getFullYear() + '-' + String(today.getMonth() + 1).padStart(2, '0') + '-' + String(today.getDate()).padStart(2, '0');
  try {
    var res = await apiFetch('/api/save-post', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        token: AUTH_TOKEN,
        headline: t.headline,
        niche: t.niche,
        platform: ap,
        contentType: contentType,
        content: output,
        scheduledDate: scheduledDate
      })
    });
    if (res.success) {
      showToast('✅ Saved!', 'success');
      await loadSchedule();
      await loadSavedPosts();
      if (TAB === 'saved') renderSaved();
    }
  } catch (e) { console.warn('Auto-save failed:', e); }
}

async function saveOutputToCalendar(tid, idx) {
  var t = ALL_TRENDS.find(function(x) { return x.id === tid; });
  if (!t) return;
  var output = OUTS[tid] && OUTS[tid][idx];
  if (!output) return;
  var ap = AP[tid] || (USER && USER.primaryPlatform) || 'instagram';
  await saveOutput(t, output, ap);
}

async function deletePost(postId) {
  try {
    var res = await apiFetch('/api/delete-post?token=' + AUTH_TOKEN + '&postId=' + postId, { method: 'DELETE' });
    if (res.success) {
      showToast('🗑️ Deleted', 'success');
      await loadSavedPosts();
      await loadSchedule();
      if (TAB === 'saved') renderSaved();
      if (TAB === 'calendar') { await loadSchedule(); renderCalendar(); }
      if (TAB === 'profile') renderProfile();
    } else {
      showToast('❌ Delete failed', 'error');
    }
  } catch (e) { showToast('❌ Error', 'error'); }
}

function openGoogleCalendar(post) {
  var title = encodeURIComponent(post.headline || 'CreatorPulse Post');
  var date = post.scheduled_date || new Date().toISOString().split('T')[0];
  var dateStr = date.replace(/-/g, '');
  var details = encodeURIComponent('Content: ' + (post.content_type || 'Script') + '\nNiche: ' + (post.niche || 'General') + '\nPlatform: ' + (post.platform || 'Unknown'));
  var url = 'https://calendar.google.com/calendar/r/eventedit?text=' + title + '&dates=' + dateStr + '/' + dateStr + '&details=' + details;
  window.open(url, '_blank');
  showToast('📅 Opening Google Calendar...', 'info');
}

function switchTab(tab) {
  TAB = tab;
  document.querySelectorAll('.page').forEach(function(p) { p.classList.remove('active'); });
  var page = document.getElementById('page-' + tab);
  if (page) page.classList.add('active');
  document.querySelectorAll('.tab-item').forEach(function(b) { b.classList.remove('active'); });
  var tabBtn = document.querySelector('.tab-item[data-tab="' + tab + '"]');
  if (tabBtn) tabBtn.classList.add('active');
  if (tab === 'home') { renderHome(); loadNotifications(); }
  if (tab === 'calendar') { loadSchedule().then(function() { renderCalendar(); }); }
  if (tab === 'saved') { loadSavedPosts().then(function() { renderSaved(); }); }
  if (tab === 'profile') { renderProfile(); }
}

var isRegistering = false;

function showAuthError(msg) {
  var el = document.getElementById('auth-error');
  if (el) {
    el.textContent = '⚠️ ' + msg;
    el.style.display = 'flex';
  }
}

function hideAuthError() {
  var el = document.getElementById('auth-error');
  if (el) el.style.display = 'none';
}

function setAuthLoading(loading) {
  var btn = document.getElementById('auth-submit');
  var txt = document.getElementById('auth-btn-text');
  if (btn) btn.disabled = loading;
  if (txt) txt.textContent = loading ? 'Loading...' : (isRegistering ? 'Create Account' : 'Sign In');
}

function toggleAuthMode() {
  isRegistering = !isRegistering;
  var confirmGroup = document.getElementById('auth-confirm-group');
  var switchText = document.getElementById('auth-switch-text');
  var switchLink = document.getElementById('auth-switch-link');
  var btnText = document.getElementById('auth-btn-text');
  if (confirmGroup) confirmGroup.style.display = isRegistering ? 'block' : 'none';
  if (switchText) switchText.textContent = isRegistering ? 'Already have an account?' : 'New here?';
  if (switchLink) switchLink.textContent = isRegistering ? 'Sign in' : 'Create account';
  if (btnText) btnText.textContent = isRegistering ? 'Create Account' : 'Sign In';
  hideAuthError();
}

async function handleAuthSubmit() {
  hideAuthError();
  var email = document.getElementById('auth-email').value.trim();
  var password = document.getElementById('auth-password').value;
  var confirm = document.getElementById('auth-confirm').value;
  if (!email || !password) { showAuthError('Please fill in all fields'); return; }
  if (isRegistering) {
    if (password !== confirm) { showAuthError('Passwords do not match'); return; }
    if (password.length < 6) { showAuthError('Password must be at least 6 characters'); return; }
  }
  setAuthLoading(true);
  try {
    if (isRegistering) {
      var name = email.split('@')[0];
      await register(email, password, name);
    } else {
      await login(email, password);
    }
    hideAuthError();
    enterApp();
  } catch (e) {
    showAuthError(e.message || 'Authentication failed');
  } finally {
    setAuthLoading(false);
  }
}

function enterApp() {
  var landing = document.getElementById('landing-screen');
  var app = document.getElementById('app');
  if (landing) landing.classList.add('hidden');
  if (app) app.style.display = 'block';
  loadAllTrends();
  loadNotifications();
  loadSchedule();
  loadSavedPosts();
  switchTab('home');
}

function initAvatarUpload() {
  var avatarInput = document.getElementById('avatar-input');
  var profileContent = document.getElementById('profile-content');
  if (!avatarInput || !profileContent) return;
  profileContent.addEventListener('click', function(e) {
    var target = e.target;
    var avatar = target.closest('#profile-avatar');
    if (avatar) {
      avatarInput.click();
    }
  });
  avatarInput.addEventListener('change', async function(e) {
    var file = e.target.files[0];
    if (!file) return;
    if (!file.type.startsWith('image/')) { showToast('❌ Please select an image file', 'error'); return; }
    if (file.size > 5 * 1024 * 1024) { showToast('❌ Image must be under 5MB', 'error'); return; }
    try {
      showToast('📤 Uploading avatar...', 'info');
      var url = await uploadAvatar(file);
      showToast('✅ Avatar updated!', 'success');
      renderProfile();
    } catch (err) {
      showToast('❌ Upload failed: ' + err.message, 'error');
    }
    avatarInput.value = '';
  });
}

function toggleNotifications() {
  NOTIF_OPEN = !NOTIF_OPEN;
  var dd = document.getElementById('notif-dd');
  if (!dd) return;
  if (NOTIF_OPEN) {
    var unreadCount = NOTIFICATIONS.filter(function(n) { return !n.read; }).length;
    var html = '<div class="notif-header"><span class="notif-header-title">🔔 Notifications</span>';
    if (unreadCount > 0) {
      html += '<button class="notif-mark-read" id="notif-mark-read">Mark all read</button>';
    }
    html += '</div>';
    if (NOTIFICATIONS.length) {
      NOTIFICATIONS.forEach(function(n) {
        html += '<div class="notif-item ' + (n.read ? 'read' : '') + '" data-nid="' + (n.id || '') + '"><div class="notif-niche">' + escapeHTML(n.niche || 'General') + '</div><div class="notif-hl">' + escapeHTML(n.headline || 'New story') + '</div><div class="notif-time">' + escapeHTML(n.time || 'Just now') + '</div></div>';
      });
    } else {
      html += '<div class="notif-empty">No notifications</div>';
    }
    dd.innerHTML = html;
    dd.classList.add('open');
    var markReadBtn = document.getElementById('notif-mark-read');
    if (markReadBtn) {
      markReadBtn.addEventListener('click', async function(e) {
        e.stopPropagation();
        await markNotificationsRead(AUTH_TOKEN);
        NOTIFICATIONS.forEach(function(n) { n.read = true; });
        var dot = document.getElementById('bell-dot');
        var btn = document.getElementById('bell-btn');
        if (dot) dot.classList.remove('show');
        if (btn) btn.classList.remove('has-unread');
        toggleNotifications();
        toggleNotifications();
      });
    }
  } else {
    dd.classList.remove('open');
  }
}

function init() {
  // Hide loading screen
  var loading = document.getElementById('loading-screen');
  if (loading) {
    setTimeout(function() { loading.classList.add('hide'); }, 800);
  }

  // Check session
  var restored = restoreSession();
  if (restored) {
    enterApp();
  } else {
    var landing = document.getElementById('landing-screen');
    if (landing) landing.classList.remove('hidden');
  }

  // Tab switching
  document.querySelectorAll('.tab-item').forEach(function(btn) {
    btn.addEventListener('click', function() { switchTab(this.dataset.tab); });
  });

  // Refresh button
  var refreshBtn = document.getElementById('refresh-btn');
  if (refreshBtn) {
    refreshBtn.addEventListener('click', function() {
      showToast('⟳ Refreshing...', 'info');
      loadAllTrends();
      loadNotifications();
    });
  }

  // Bell notifications
  var bellBtn = document.getElementById('bell-btn');
  var bellWrap = document.getElementById('bell-wrap');
  if (bellBtn) {
    bellBtn.addEventListener('click', function(e) {
      e.stopPropagation();
      toggleNotifications();
    });
  }
  document.addEventListener('click', function(e) {
    if (!e.target.closest('#bell-wrap')) {
      var dd = document.getElementById('notif-dd');
      if (dd) { dd.classList.remove('open'); NOTIF_OPEN = false; }
    }
  });

  // Logout
  var logoutBtn = document.getElementById('logout-btn');
  if (logoutBtn) {
    logoutBtn.addEventListener('click', function() {
      if (confirm('Log out?')) { clearSession(); location.reload(); }
    });
  }

  // Auth form
  var authSubmit = document.getElementById('auth-submit');
  var authSwitch = document.getElementById('auth-switch-link');
  if (authSubmit) authSubmit.addEventListener('click', handleAuthSubmit);
  if (authSwitch) authSwitch.addEventListener('click', toggleAuthMode);

  // Enter key on auth inputs
  ['auth-email', 'auth-password', 'auth-confirm'].forEach(function(id) {
    var el = document.getElementById(id);
    if (el) {
      el.addEventListener('keypress', function(e) {
        if (e.key === 'Enter') handleAuthSubmit();
      });
    }
  });

  // Event delegation
  initEventDelegation();

  // Avatar upload
  initAvatarUpload();

  // Modal close buttons
  var imageModalClose = document.getElementById('image-modal-close');
  if (imageModalClose) imageModalClose.addEventListener('click', closeImageModal);
  var dayDetailClose = document.getElementById('day-detail-close');
  if (dayDetailClose) dayDetailClose.addEventListener('click', closeDayDetail);
  var dayDetailOverlay = document.getElementById('day-detail-overlay');
  if (dayDetailOverlay) dayDetailOverlay.addEventListener('click', closeDayDetail);

  // Auto-refresh every 30 minutes
  setInterval(function() {
    if (AUTH_MODE === 'app') {
      loadAllTrends();
      loadNotifications();
      if (TAB === 'calendar') { loadSchedule().then(function() { renderCalendar(); }); }
      if (TAB === 'saved') loadSavedPosts().then(function() { renderSaved(); });
    }
  }, 30 * 60 * 1000);
}
