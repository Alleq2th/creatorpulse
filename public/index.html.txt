async function loadSubjectImage(headline, niche, articleImageUrl) {
  if (articleImageUrl) {
    const proxiedUrl = `${API}/api/image-proxy?url=${encodeURIComponent(articleImageUrl)}`;
    const realImg = await loadImageElement(proxiedUrl);
    if (realImg) return realImg;
  }

  const sportKeywords = ["football","soccer","basketball","tennis","cricket","boxing","mma","nba","nfl","premier league","champions league","f1"];
  const isSport = sportKeywords.some(k => (niche+headline).toLowerCase().includes(k));

  const words = headline.split(" ");
  const properNouns = words.filter(w => w.length > 2 && w[0] === w[0].toUpperCase() && w[0] !== w[0].toLowerCase());
  const subject = properNouns.slice(0,2).join(" ");

  if (!subject) return null;

  try {
    let imgUrl = null;
    if (isSport) {
      imgUrl = await fetchSportsImage(subject, "player");
      if (!imgUrl) imgUrl = await fetchSportsImage(subject, "team");
    }
    if (!imgUrl) imgUrl = await fetchWikiImage(subject);
    if (!imgUrl) return null;
    return await loadImageElement(imgUrl);
  } catch(e) { return null; }
}

function loadImageElement(url) {
  return new Promise((resolve) => {
    const img = new Image();
    img.crossOrigin = "anonymous";
    img.onload = () => resolve(img);
    img.onerror = () => resolve(null);
    img.src = url;
    setTimeout(() => resolve(null), 6000);
  });
}

function extractJSON(text) {
  const match = text.match(/\{[\s\S]*\}/);
  return match ? JSON.parse(match[0]) : null;
}

// Restore session from localStorage
function restoreSession() {
  try {
    const saved = localStorage.getItem("cp_session");
    if (saved) {
      const { token, user } = JSON.parse(saved);
      if (token && user) {
        AUTH_TOKEN = token;
        USER = user;
        AUTH_MODE = "app";
        return true;
      }
    }
  } catch(e) {}
  return false;
}

function saveSession(token, user) {
  localStorage.setItem("cp_session", JSON.stringify({ token, user }));
}

function clearSession() {

  AUTH_TOKEN = null;
  USER = null;
  AUTH_MODE = "landing";

function renderCal(){

  const first=new Date(CAL_Y,CAL_M,1).getDay(),days=new Date(CAL_Y,CAL_M+1,0).getDate();
  
  // Group posts by date
  const postsByDate = {};
  SCHEDULE_DATA.forEach(post => {
    if (post.scheduled_date) {
      const day = new Date(post.scheduled_date).getDate();
      if (!postsByDate[day]) postsByDate[day] = [];
      postsByDate[day].push(post);
    }
  });

  const em={3:[{t:"post",x:"Post due"}],6:[{t:"event",x:"Niche event"}],10:[{t:"reminder",x:"Reminder"}],14:[{t:"post",x:"Mid-week post"},{t:"event",x:"Industry event"}],17:[{t:"reminder",x:"Weekly recap"}],21:[{t:"event",x:"Major event"}],25:[{t:"post",x:"End of week"}]};
  if(isCur)em[today]=[{t:"reminder",x:"Post today!"},{t:"post",x:"Trending now"}];
  const cells=[];for(let i=0;i<first;i++)cells.push(null);for(let d=1;d<=days;d++)cells.push(d);

    const countBadge = dayPosts.length > 0 ? `<div style="font-size:8px;color:var(--pl);margin-top:2px">📌 ${dayPosts.length}</div>` : "";
    const hasPost = dayPosts.length > 0 ? 'has-post' : '';
    return`<div class="cc ${isTod?"td":""} ${hasPost}" style="${dayPosts.length > 0 ? 'background:rgba(124,58,237,.05)' : ''}"><div class="cdt ${isTod?"tod":""}">${d}</div>${evs}${countBadge}</div>`;

  const mo=MONTHS[CAL_M];
  const up=[

// ── BIND APP ──────────────────────────────────────────────────────────────────
function updateSingleCard(tid){
  const t = ALL_TRENDS.find(x=>x.id===tid);
  if(!t) return;
  const oldEl = document.getElementById("tc-"+tid);
  if(!oldEl){ render(); bindApp(); return; }
  const wrap = document.createElement("div");
  wrap.innerHTML = renderTC(t);
  const newEl = wrap.firstElementChild;
  oldEl.replaceWith(newEl);
  bindCard(newEl, tid);
}

function bindCard(cardEl, tid){
  cardEl.querySelector(".th")?.addEventListener("click", async ()=>{
    EXP[tid]=!EXP[tid];
    if(EXP[tid]&&!AP[tid])AP[tid]=USER.primaryPlatform;
    if(EXP[tid]&&!AT[tid])AT[tid]=PTYPES[USER.primaryPlatform]?.[0]||"";
    updateSingleCard(tid);
    if(EXP[tid]&&!RECS[tid]){
      const t=ALL_TRENDS.find(x=>x.id===tid);
      if(t){const rec=await generateText("Give a one-sentence content recommendation. Direct, specific, no preamble.",`Niches: ${USER.niches.join(", ")}. Story: "${t.headline}". Best platform and format?`);RECS[tid]=rec;updateSingleCard(tid);}
    }
  });
  cardEl.querySelectorAll(".ptab[data-tid]").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();AP[tid]=el.dataset.pid;AT[tid]=PTYPES[el.dataset.pid]?.[0]||"";updateSingleCard(tid);}));
  cardEl.querySelectorAll(".ctp[data-tid]").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();AT[tid]=el.dataset.ct;updateSingleCard(tid);}));
  cardEl.querySelectorAll(".gen-btn").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();gen(tid,false);}));
  cardEl.querySelectorAll(".gen-all-btn").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();gen(tid,true);}));
  cardEl.querySelectorAll(".cb[data-copy]").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();navigator.clipboard.writeText(decodeURIComponent(el.dataset.copy));el.textContent="✓ Copied";el.classList.add("copied");setTimeout(()=>{el.textContent="Copy";el.classList.remove("copied");},2000);}));
  cardEl.querySelectorAll(".dl-btn").forEach(el=>el.addEventListener("click",e=>{e.stopPropagation();const a=document.createElement("a");a.href=el.dataset.url;a.download=el.dataset.name;document.body.appendChild(a);a.click();document.body.removeChild(a);}));
}

function bindApp(){
  document.querySelectorAll(".ni[data-tab]").forEach(el=>el.addEventListener("click",()=>{TAB=el.dataset.tab;render();bindApp();if(TAB==="calendar")loadSchedule();}));

async function gen(tid, all) {
  const t=ALL_TRENDS.find(x=>x.id===tid);if(!t)return;
  const ap=AP[tid]||USER.primaryPlatform,at=AT[tid]||PTYPES[ap]?.[0]||"",accent=gc(t.niche);
  LOAD[tid]=true;OUTS[tid]=[];updateSingleCard(tid);
  try{
    const types=all?(PTYPES[ap]||[]):[at];const res=[];
    const subjectImg = await loadSubjectImage(t.headline, t.niche, t.image);

    for(const ct of types){
      if(ct==="IG Carousel"){
        let sc=5;
        try{
          const cr=await generateText('Return ONLY valid JSON: {"slideCount":number}',`Story: "${t.headline}". Recommend Instagram carousel slide count 3-7.`);
          const parsed = extractJSON(cr);
          if(parsed) sc=parsed.slideCount||5;
        }catch{}
        let slides=[];
        try{
          const sr=await generateText(`Return ONLY a valid JSON array of ${sc} objects: [{"slideNum":1,"title":"bold hook","body":"2-3 punchy sentences"}]. No markdown.`,`Create a ${sc}-slide viral Instagram carousel for ${t.niche} creator. Story: "${t.headline}". Summary: ${t.summary}`);
          const parsed = extractJSON(sr);
          if(parsed) slides = Array.isArray(parsed) ? parsed : [parsed];
          else slides = JSON.parse(sr);
        } catch{ slides=[{slideNum:1,title:t.headline,body:t.summary||""}]; }
        const images=slides.map((s,i)=>makeGraphicCanvas(s.title||t.headline,subjectImg,i+1,slides.length,accent,USER.name));
        const cap=await generateText(SYS,`Instagram caption + 10 hashtags for ${t.niche} carousel: "${t.headline}". Punchy, ready to post.`);
        res.push({type:"carousel",slides,images});res.push({type:"text",label:"📝 Caption + Hashtags",content:cap});
      }else if(ct==="Thumbnail"){
        let hl=t.headline.toUpperCase().split(" ").slice(0,5).join(" ");
        try{
          const tr=await generateText('Return ONLY valid JSON: {"headline":"YOUR HEADLINE HERE"}. The headline must be max 6 words, ALL CAPS, directly about the story topic, shocking or curiosity-driving. Think MrBeast thumbnail energy. No generic words like "SHOCKING" or "CRAZY" — be specific to the actual story.',`Niche: ${t.niche}. Story: "${t.headline}". Write the perfect YouTube thumbnail headline for THIS specific story.`);
          const parsed = extractJSON(tr);
          if(parsed && parsed.headline) hl=parsed.headline;
        }catch{}
        res.push({type:"thumbnail",img:makeThumbnailCanvas(hl,subjectImg,USER.name,accent)});
      }else if(ct==="TikTok No-Face Graphic"){
        const hook=await generateText("You are a viral TikTok hook writer. Return ONLY the hook text — max 6 words, ALL CAPS, no punctuation except exclamation marks. The hook must directly reference the actual story topic. Make it shocking, curiosity-driving, or emotional. No random unrelated hooks.",`Story: "${t.headline}". Niche: ${t.niche}. Write the perfect TikTok text hook for this specific story.`);
        const cap=await generateText(SYS,`TikTok caption + 6 hashtags for ${t.niche}: "${t.headline}". Ready to post.`);
        res.push({type:"noface",hook:hook.trim(),caption:cap,img:makeGraphicCanvas(hook.trim()||t.headline,subjectImg,1,1,accent,USER.name)});
      }else if(ct==="TikTok Script"){
        let ld={length:"short",targetSeconds:30,reason:"Quick punchy take"};
        try{
          const lr=await generateText('Return ONLY valid JSON: {"length":"short","targetSeconds":30,"reason":"one sentence"}',`${t.niche}: "${t.headline}". Best TikTok length?`);
          const parsed = extractJSON(lr);
          if(parsed) ld = parsed;
        }catch{}
        const s=await generateText(SYS,`Write a ${ld.length==="long"?"60-90 second":"15-30 second"} TikTok script for a ${t.niche} creator about: "${t.headline}". Write it as continuous spoken narration exactly as the creator would say it on camera, word for word, ready for a teleprompter. Start with the hook line, flow naturally into the body, end with a call to action. Do not use headers, bullets, or labels — just the words to say, as flowing natural speech with paragraph breaks where the creator would naturally pause.`);
        res.push({type:"text",label:`🎵 TikTok Script (${ld.length==="long"?"Long":"Short"} ~${ld.targetSeconds}s) — ${ld.reason}`,content:s});
      }else if(ct==="Documentary Script"){
        const s=await generateText(SYS,`Write a YouTube documentary narration script for a ${t.niche} creator about: "${t.headline}". Write it as continuous spoken narration exactly as a narrator would read it aloud, word for word, ready for a teleprompter, roughly 700-800 words. It should flow as one continuous voiceover covering the opening hook, the background story, the main conflict or development, an analysis of what it means, and a closing thought with a call to action — but written as natural spoken paragraphs, not labeled sections. You may include occasional bracketed visual notes like [show footage of X] sparingly, but the majority of the text must be pure spoken narration.`);
        res.push({type:"text",label:"🎬 Documentary Script",content:s});
      }else if(ct==="Long Form Script"){
        const s=await generateText(SYS,`Write a full YouTube video script for a ${t.niche} creator about: "${t.headline}". Write it as continuous spoken narration exactly as the creator would say it on camera, word for word, ready for a teleprompter, roughly 700-900 words. Open with a strong hook, introduce the topic naturally, cover the main points as flowing spoken paragraphs, and close with a call to action. Do not use headers, bold text, bullet points, or section labels of any kind — just the natural words to say out loud, broken into paragraphs where a speaker would naturally pause for breath or transition.`);
        res.push({type:"text",label:"▶️ Long Form Script",content:s});
      }else if(ct==="YouTube Short Script"){
        const s=await generateText(SYS,`Write a 45-60 second YouTube Shorts script for a ${t.niche} creator about: "${t.headline}". Write it as continuous spoken narration exactly as the creator would say it on camera, word for word, ready for a teleprompter. Open with a 3-second hook that stops the scroll, deliver the core point fast and punchy, end with a strong call to action. Plain spoken paragraphs only — no headers, no JSON, no bullet points, no labels.`);
        res.push({type:"text",label:"▶️ YouTube Short Script",content:s});
      }else if(ct==="Title + Description"){
        const s=await generateText(SYS,`3 YouTube titles + SEO description for ${t.niche}. Topic: "${t.headline}". Titles: clickbait. Description: keywords + hashtags.`);
        res.push({type:"text",label:"▶️ Title + Description",content:s});
      }else if(ct==="Thread (5 tweets)"){
        const s=await generateText(SYS,`Write a viral 5-tweet X thread for a ${t.niche} creator about: "${t.headline}". Format each line exactly as "Tweet 1/5: [text]" through "Tweet 5/5: [text]", max 280 characters each, punchy and natural. Last tweet ends with a call to action. Return ONLY the five tweet lines as plain text — no JSON, no markdown, no extra commentary.`);
        res.push({type:"text",label:"𝕏 Thread",content:s});
      }else if(ct==="IG Reel Script"){
        const s=await generateText(SYS,`Write a 30-60 second Instagram Reel script for a ${t.niche} creator about: "${t.headline}". Write it as continuous spoken narration exactly as the creator would say it on camera, word for word, ready for a teleprompter. Open with a hook, deliver the point, close with a call to action. Plain natural spoken paragraphs only — no headers, no labels, no bullet points.`);
        res.push({type:"text",label:"📸 IG Reel Script",content:s});
      }else{
        const p=PLATS.find(x=>x.id===ap);
        const s=await generateText(SYS,`Write a "${ct}" for a ${t.niche} creator on ${p?.label||ap} about: "${t.headline}". ${t.summary}. Return ONLY the plain ready-to-post text itself — no JSON, no curly braces, no key-value wrapping, no markdown formatting, no quotation marks around the whole thing. Just the raw text exactly as it should be pasted into the post.`);
        res.push({type:"text",label:`${p?.icon||""} ${ct}`,content:s});
      }
    }
    OUTS[tid]=res;
    
    // ✅ AUTO-SAVE to calendar
    if (res.length > 0 && USER && AUTH_TOKEN) {
      const firstPiece = res[0];
      let contentType = "Script";
      if (firstPiece.type === "carousel") contentType = "Carousel";
      else if (firstPiece.type === "thumbnail") contentType = "Thumbnail";
      else if (firstPiece.type === "noface") contentType = "Graphic";
      
      try {
        await fetch(`${API}/api/save-post`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            token: AUTH_TOKEN,
            headline: t.headline,
            niche: t.niche,
            platform: ap,
            contentType: contentType,
            content: firstPiece,
            scheduledDate: new Date().toISOString().split('T')[0]
          })
        });
        console.log("✅ Post auto-saved to calendar!");
        // Refresh schedule data if we're on calendar tab
        if (TAB === "calendar") loadSchedule();
      } catch(e) { console.log("Save error:", e); }
    }
    
  }catch(e){OUTS[tid]=[{type:"text",label:"Error",content:"Generation failed: "+(e.message||"Unknown error")+"\n\nCheck that your Groq API key is set correctly in Render's environment variables."}];}
  LOAD[tid]=false;updateSingleCard(tid);
}
